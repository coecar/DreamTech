3   import java.sql.PreparedStatement;
4   import java.sql.ResultSet;
5   import java.sql.SQLException;
6   import java.util.ArrayList;
7   import java.util.List;

15  public class GestorCanciones {
16  	/**
17  	 * Variable que almacena el agente de la base de datos.
18  	 */
19  	private Agente agenteBD;
20  	/**
21  	 * Variable que almacena todas las canciones.
22  	 */
23  	private List<Cancion> canciones;
24  	/**
25  	 * Metodo constructor de la clase.
26  	 * @throws ClassNotFoundException Excepción de clase.
27  	 */
28  	public GestorCanciones() throws ClassNotFoundException {
29  		canciones = new ArrayList<Cancion>();
30  		agenteBD = Agente.getAgente();
31  	}
32  	/**
33  	 * Metodo usado para leer una canción de la base de datos.
34  	 * @param Nombre de la canción a leer.
35  	 * @return cancion leida.
36  	 * @throws SQLException Excepción de base de datos.
37  	 * @throws ClassNotFoundException Excepción de clase.
38  	 */
39  	public Cancion read(String Nombre) throws SQLException, ClassNotFoundException {
40  		String sql = "SELECT * FROM usuarios WHERE Nick = '" + Nombre + "';";
41  		String Artista = null;
42  		double Precio = 0;
43  		ResultSet rs;
44  		rs = agenteBD.read(sql);
45  		if (rs.next()) {
46  			Nombre = rs.getString("Nombre");
47  			Artista = rs.getString("Artista");
48  			Precio = rs.getDouble("Precio");
49  		}
50  		return new Cancion(Nombre, Artista, Precio);
51  	}
52  	/**
53  	 * Metodo para leer todas las canciones de la base de datos.
54  	 * @return la lista de canciones de la base de datos.
55  	 * @throws SQLException Excepción de base de datos.
56  	 * @throws ClassNotFoundException Excepción de clase.
57  	 */
58  	public List<Cancion> readAll() throws SQLException, ClassNotFoundException {
59  		String sql = "SELECT * FROM usuarios";
60  		String Nombre = null, Artista = null;
61  		Double Precio;
62  		ResultSet rs;
63  		rs = agenteBD.read(sql);
64  		while (rs.next()) {
65  			Nombre = rs.getString("Nombre");
66  			Artista = rs.getString("Artista");
67  			Precio = rs.getDouble("Precio");
68  			canciones.add(new Cancion(Nombre, Artista, Precio));
69  		}
70  		return canciones;
71  	}
72  	/**
73  	 * Metodo para insertar una cancion en la base de datos.
74  	 * @param cancion que se quiere insertar en la base de datos.
75  	 * @return resultado un entero que dice si se ha insertado correctamente.
76  	 * @throws SQLException Excepción de base de datos.
77  	 */
78  	public int insert(Cancion cancion) throws SQLException {
79  		String sql = "INSERT INTO usuarios(Nick,Nombre,Apellido,Password) VALUES(?,?,?,?)";
80  		PreparedStatement pst = agenteBD.getConection().prepareStatement(sql);
81  		pst.setString(1, cancion.getNombre());
82  		pst.setString(2, cancion.getArtista());
83  		pst.setDouble(3, cancion.getPrecio());
84  
85  		int resultado = agenteBD.insert(pst);
86  		return resultado;
87  	}
88  	/**
89  	 * Metodo para eliminar una cancion de la base de datos.
90  	 * @param cancion que se quiere eliminar de la base de datos.
91  	 * @return resultado Entero que dice si se ha eliminado correctamente.
92  	 * @throws SQLException Excepción de base de datos.
93  	 */
94  	public int delete(Cancion cancion) throws SQLException {
95  		String sql = "DELETE FROM canciones WHERE Nombre=" + cancion.getNombre();
96  		agenteBD.getConection().prepareStatement(sql);
97  		int resultado = agenteBD.delete(sql); 
98  		return resultado;
99  	}

101 }
